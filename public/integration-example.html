// 1. 监听iframe的消息
window.addEventListener('message', (event) => {
  const { type, payload } = event.data;

  // iframe请求数据
  if (type === 'PRINT_PAGE_REQUEST_DATA') {
    console.log('iframe请求数据:', payload);

    // 2. 获取当前页面的通用数据
    const currentData = getCurrentData(); // 你的业务数据

    // 3. 发送给iframe
    event.source.postMessage({
      type: 'PRINT_DATA',
      payload: currentData
    }, '*');
  }

  // iframe数据接收确认
  if (type === 'PRINT_DATA_RECEIVED') {
    console.log('iframe已接收数据:', payload);
  }
});

// 4. 关闭iframe时清除缓存
function closeIframe() {
  iframe.contentWindow.postMessage({
    type: 'CLEAR_DATA',
    payload: { reason: 'parent_close' }
  }, '*');
}

// 5. 打开iframe（会自动请求数据）
function openIframe() {
  iframe.src = 'http://localhost:3000/print';
}
